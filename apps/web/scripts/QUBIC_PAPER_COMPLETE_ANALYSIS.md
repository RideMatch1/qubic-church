// Aus Qubic Core - score_addition.h
template <unsigned long long bitCount>
static void toTenaryBits(long long A, char* bits) {
    for (unsigned long long i = 0; i < bitCount; ++i) {

        bits[i] = (bitValue == 0) ? -1 : bitValue;  // 0→-1, 1→+1
