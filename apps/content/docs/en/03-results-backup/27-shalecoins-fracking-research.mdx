import hashlib
import base58
from ecdsa import SigningKey, SECP256k1

def pawgo_method(tx_id):
    """
    PawGo's proven exploitation method

    Input: Transaction ID from blockchain
    Output: Bitcoin address (if it was used as seed)
    """

    # Step 1: Hash the TX ID as seed
    seed = tx_id.encode()
    privkey_bytes = hashlib.sha256(seed).digest()

    # Step 2: Derive private key
    privkey = SigningKey.from_string(privkey_bytes, curve=SECP256k1)

    # Step 3: Generate public key
    pubkey = privkey.get_verifying_key()
    x, y = pubkey.pubkey.point.x(), pubkey.pubkey.point.y()

    # Step 4: Compress public key
    pubkey_compressed = (
        (b'\x02' if y % 2 == 0 else b'\x03') +
        x.to_bytes(32, 'big')

    # Step 5: Hash to address
    hash160 = hashlib.new('ripemd160',
                         hashlib.sha256(pubkey_compressed).digest()

    versioned = b'\x00' + hash160
    checksum = hashlib.sha256(
                   hashlib.sha256(versioned).digest()
               ).digest()[:4]

    # Step 7: Base58 encode
    address = base58.b58encode(versioned + checksum).decode()

    return address
