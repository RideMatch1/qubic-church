def seed_to_matrix_coordinates(seed: str) -> tuple[int, int]:
    """
    Qubic Seed → Matrix Coordinates

    Algorithm:
    1. UTF-8 encode the seed
    2. SHA256 hash
    3. Extract first two bytes
    4. Apply modulo 128
    """
    h = hashlib.sha256(seed.encode('utf-8')).digest()

    col = h[1] % 128
    return (row, col)

def matrix_to_anna(row: int, col: int) -> tuple[int, int]:

    Matrix indices → Anna coordinates
